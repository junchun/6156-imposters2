  //check game board state
  public int checkGB(int x, int y)
  {
	int x_coord = this.boardState.length;
	int y_coord = this.boardState[0].length; 
	
	if (x>=x_coord || y>=y_coord)
	{
		return -1;
	} else
	{
		if (this.boardState[x][y] == 'O' || this.boardState[x][y] == 'X')
		{
			return 0;
		} else
		{
			return 1;
		}
	}
  }
