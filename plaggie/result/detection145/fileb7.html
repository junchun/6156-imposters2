package models;

public class GameBoard {

  /**
   * The object representation of the game state.
   *
   * @param p1 The first player
   * @param p2 The second player
   * @param gameStarted Whether both players have joined
   * @param turn Which player's turn it is
   * @param boardState The board representation
   * @param winner Whether there is a winner
   * @param isDraw Whether there is a draw
   */

  public GameBoard(
      Player p1,
      Player p2,
      boolean gameStarted,
      int turn,
      char[][] boardState,
      int winner,
      boolean isDraw) {
    super();
    this.p1 = p1;
    this.p2 = p2;
    this.gameStarted = gameStarted;
    this.turn = turn;
    this.boardState = boardState;
    this.winner = winner;
    this.isDraw = isDraw;
  }

  private Player p1;

  private Player p2;

  private boolean gameStarted;

  private int turn;

  private char[][] boardState;

  private int winner;

  private boolean isDraw;

    return p2;
  }

  }

  public boolean isGameStarted() {
    return gameStarted;

    this.gameStarted = gameStarted;
